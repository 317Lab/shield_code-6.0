229void FSMUpdate(){
230 switch(currentState){
231 case idle: {
232 if (syncPulse) {
233 syncPulse = false;
234 return;
235 }
236 volatile uint32_t irq_state = __get_PRIMASK();
237 __disable_irq();
238 if (micros() - timer > SWEEP_OFFSET) {
239 if (isFirst){
240 isFirst = false;
241 timer = micros();
242 newCycle = false;
243 }
244 currentState = startSweep;
245 }
246 __set_PRIMASK(irq_state);
247 break;
248 }
249
250 case startSweep: {
251 if (syncPulse) {
252 currentState = interrupted;
253 syncPulse = false;
254 } else {
255 currentState = takeIMU;
256 }
257 break;
258 }
259
260 case takeIMU: {
261 if (syncPulse) {
262 currentState = interrupted;
263 syncPulse = false;
264 } else {
265 currentState = read;
266 }
267 break;
268 }
269 case read:
270 if (syncPulse) {
271 currentState = interrupted;
272 syncPulse = false;
273 } else {
274 currentState = store;
275 }
276 break;
277 case store: {
278 if (syncPulse) {
279 currentState = interrupted;
280 syncPulse = false;
281 } else {
282 currentState = waitForNewCycle;
283 }
284 break;
285 }
286
287 case waitForNewCycle: {
288 if (newCycle || syncPulse) {
289 currentState = idle;
290 newCycle = false;
291 syncPulse = false;
292 }
293 break;
294 }
295 case interrupted: {
296 currentState = waitForNewCycle;
297 break;
298 }
299 default: {
300 currentState = idle;
301 break;
302 }

425void configureTimerInterrupt(){
426 // Enable the clock to the TC0 peripheral
427 pmc_enable_periph_clk(ID_TC0);
428
429 /*Configures the timer:
430 - First two parameters set it to RC compare waveform mode. This means the timer resets when it reaches the value in RC.
431 - The third parameter sets the clock source to MCK/128. MCK is at 84 MHz, so this sets the clock to 656.25 kHz.
432 */
433 TC_Configure(TC0, 0, TC_CMR_WAVE | TC_CMR_WAVSEL_UP_RC | TC_CMR_TCCLKS_TIMER_CLOCK4);
434 //RC sets the value that the counter reaches before triggering the interrupt
435 //This sets it to 10kHz
436 TC_SetRC(TC0, 0, 64.625); //
437
438 // Enable the interrupt RC compare interrupt
439 TC0->TC_CHANNEL[0].TC_IER = TC_IER_CPCS;
440 // Disable all other TC0 interrupts
441 TC0->TC_CHANNEL[0].TC_IDR = ~TC_IER_CPCS;
442 NVIC_EnableIRQ(TC0_IRQn);
443 NVIC_SetPriority(TC0_IRQn, 0);
444
445 TC_Start(TC0, 0);
446}

503 sendFromRam = true;
